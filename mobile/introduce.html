# 注意

  • 安装 test 单元会报错，先设置源
npm config set chromedriver_cdnurl=http://cdn.npm.taobao.org/dist/chromedriver
yarn config set "chromedriver_cdnurl" "https://npm.taobao.org/mirrors/chromedriver"

# 命令

// 项目安装
yarn install
// 启动开发环境
yarn run serve
// 打包
yarn build
// eslint 检查
yarn run lint
// stylelint 检查并自动修复
yarn run lint:style
// 单元测试
yarn run test:unit
// e2e测试
yarn run test:e2e

# 目录结构

├── ./README.md
├── ./babel.config.js                             // babel 配置文件
├── ./jest.config.js                              // 单元测试 jest 配置文件
├── ./package-lock.json
├── ./package.json
├── ./postcss.config.js                           // postcss 配置文件
├── ./public
│   ├── ./public/favicon.ico
│   └── ./public/index.html
├── ./src
│   ├── ./src/App.vue                               // 入口文件
│   ├── ./src/api                                   // 所有请求
│   ├── ./src/assets                                // 静态资源
│   ├── ./src/components                            // 组件
│   ├── ./src/main.js                               // 入口文件  初始化等
│   ├── ./src/mixins                                // 混合
│   │   ├── ./src/mixins/routeBack.js               // 返回
│   │   └── ./src/mixins/savePosition.js            // 保存滚动位置
│   ├── ./src/plugins                               // 插件
│   │   ├── ./src/plugins/axios.js
│   │   ├── ./src/plugins/fs-ui.js                  // fs-ui所使用到的组件和插件
│   │   ├── ./src/plugins/rem.js                    // 自适应
│   │   ├── ./src/plugins/touch.js                  // 触摸指令
│   │   ├── ./src/plugins/vconsole.js               // vconsole
│   │   └── ./src/plugins/vueg.js                   // 转场动画
│   ├── ./src/router                                // 路由存放
│   │   └── ./src/router/index.js
│   ├── ./src/store
│   │   ├── ./src/store/modules
│   │   └── ./src/store/index.js
│   ├── ./src/untils
│   └── ./src/views
│   │    ├── ./src/views/Root.vue                    // 布局文件
├── .stylelintignore                                 // stylelint 忽略项
├── ./stylelint.config.js                            // stylelint 配置项
├── ./tests
├── ./vendor-manifest.json
├── ./vue.config.js
├── ./webpack.dll.config.js
└── ./yarn.lock
